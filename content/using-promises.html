{% localize using_promises.1 %}

{% highlight js %} function loadImageWithCallback(src, callback) { // call callback in the future when the image loads } loadImageWithCallback('myimage.png', function(img) { // img loaded! }); {% endhighlight %}

{% localize using_promises.with_a_promise %}

{% highlight js %} function loadImagePromise(src) { // return a promise of the future loaded image } loadImagePromise('myimage.png').then(function(img) { // img loaded! }); {% endhighlight %}

{% localize using_promises.can_call_later %}

{% highlight js %} var imagePromise = loadImagePromise('myimage.png'); // some time has passed and the image has completed loading imagePromise.then(function(img) { // this still gets called }); {% endhighlight %}

{% localize using_promises.call_multiple_times %}

{% highlight js %} var imagePromise = loadImagePromise('myimage.png'); imagePromise.then(function(img) { // do something with the image }); // somewhere else in the code imagePromise.then(function(img) { // do something else with the image }); {% endhighlight %}

{% localize using_promises.configure_later %}

{% highlight js %} var imagePromise = loadImagePromise('myimage.png'); // somewhere else in the code getImagePromise().then(function(img) { // when the image loads. }); {% endhighlight %}
